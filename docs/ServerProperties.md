<h3 align="center">--==[ server.properties example ]==--</h3>

```properties
# Server connection
# Open server to LAN mode like player
open-lan = false
server-ip = 0.0.0.0
server-port = 25565
online-mode = false

# Server locale
# ./locales/default.properties
server-locale = default

# Debug messages etc
debug-mode = false

# WhiteList
enable-whitelist = false
# uuid | name
whitelist-mode = uuid

# BanList
enable-banlist = false
# uuid | name | ip
banlist-mode = uuid

```
